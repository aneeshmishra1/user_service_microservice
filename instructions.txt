
user table in database - my-first-postegresql-db
CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    email VARCHAR UNIQUE,
    username VARCHAR UNIQUE,
    first_name VARCHAR,
    last_name VARCHAR,
    hashed_password VARCHAR,
    is_active BOOLEAN DEFAULT TRUE,
    roles VARCHAR
);

todo-table in database - my-second-postgresql-db
CREATE TABLE todos (
    id SERIAL PRIMARY KEY,
    title VARCHAR,
    description VARCHAR,
    priority INTEGER,
    completed BOOLEAN DEFAULT FALSE,
    owner_id INTEGER
);
